### Task 1: Create `run_forecast_dec2025.py`

**Files:**
- Create: `scripts/run_forecast_dec2025.py`

**Step 1: Write the script**

This script finds the best rolling model per market (from rolling_accuracy.csv), trains it on all data through Nov 2025, and produces a 1-step Dec 2025 forecast.

```python
#!/usr/bin/env python3
"""Dec 2025 Forecast — 1-step rolling forecast for each market.

Picks the best rolling model per series (lowest median APE from rolling
accuracy results), retrains on all available data (through Nov 2025),
and forecasts Dec 2025.

Output (to reports/):
  {direction}_forecast_dec2025.csv

Usage:
  python scripts/run_forecast_dec2025.py --direction inbound
  python scripts/run_forecast_dec2025.py --direction outbound
  python scripts/run_forecast_dec2025.py  # both
"""
from __future__ import annotations

import argparse
import logging
import os
import time
import warnings
from pathlib import Path

logging.disable(logging.CRITICAL)
os.environ["CMDSTAN_VERBOSE"] = "false"

import numpy as np
import pandas as pd

from config import (
    CALL_TYPES, DASHBOARD_CALL_TYPES,
    grain_cols_for, target_for,
)
from models import (
    forecast_seasonal_naive_1step, forecast_ets_damped_1step,
    forecast_theta_1step,
)

ROOT = Path(__file__).resolve().parent.parent


def parse_args() -> argparse.Namespace:
    p = argparse.ArgumentParser(description="Dec 2025 rolling forecast.")
    p.add_argument("--direction", choices=["inbound", "outbound"],
                   default=None, help="Direction (omit for both)")
    return p.parse_args()


ROLLING_MODELS = {
    "ets_damped": forecast_ets_damped_1step,
    "theta": forecast_theta_1step,
}


def run_direction(direction: str):
    gc = grain_cols_for(direction)
    reports = ROOT / "reports"

    # Load rolling accuracy to find best model per series
    rolling_path = reports / f"{direction}_rolling_accuracy.csv"
    rolling = pd.read_csv(rolling_path)
    rolling = rolling[rolling["CALL_TYPE"].isin(DASHBOARD_CALL_TYPES)]
    # Only non-fallback, non-lgbm models (ets_damped, theta)
    rolling = rolling[rolling["model"].isin(ROLLING_MODELS.keys())]

    # Best rolling model per series = lowest median APE
    group_cols = gc + ["target", "model"]
    median_ape = rolling.groupby(group_cols)["ape"].median().reset_index()
    median_ape.columns = gc + ["target", "model", "median_ape"]
    idx = median_ape.groupby(gc + ["target"])["median_ape"].idxmin()
    best_rolling = median_ape.loc[idx].reset_index(drop=True)

    # Load actual data
    df = pd.read_parquet(ROOT / "data" / f"{direction}_set.parquet")
    df = df[df["CALL_TYPE"].isin(DASHBOARD_CALL_TYPES)]
    df = df.sort_values(gc + ["CALL_YEAR_MONTH"]).reset_index(drop=True)
    grouped = {k: g.sort_values("CALL_YEAR_MONTH") for k, g in df.groupby(gc)}

    rows = []
    n = len(best_rolling)
    t0 = time.time()
    fallbacks = 0

    for i, (_, winner) in enumerate(best_rolling.iterrows()):
        key = tuple(winner[c] for c in gc)
        target = winner["target"]
        model_name = winner["model"]

        if key not in grouped:
            continue

        series = grouped[key]
        train_vals = series[target].values.astype(float)

        if len(train_vals) < 12:
            continue

        forecast_fn = ROLLING_MODELS.get(model_name)
        pred = forecast_fn(train_vals) if forecast_fn else None

        used = model_name
        if pred is None:
            pred = forecast_seasonal_naive_1step(train_vals)
            used = f"{model_name}_fb"
            fallbacks += 1

        base = {c: winner[c] for c in gc}
        rows.append({
            **base,
            "CALL_YEAR_MONTH": 202512,
            "target": target,
            "model": used,
            "predicted": pred,
        })

        if (i + 1) % 500 == 0:
            el = time.time() - t0
            r = (i + 1) / el
            print(f"  {i+1}/{n} ({el:.0f}s, {r:.1f}/s)", flush=True)

    el = time.time() - t0
    print(f"  {direction}: {len(rows)} forecasts in {el:.0f}s ({fallbacks} fallbacks)", flush=True)

    out_path = reports / f"{direction}_forecast_dec2025.csv"
    pd.DataFrame(rows).to_csv(out_path, index=False)
    print(f"  Saved: {out_path}", flush=True)


def main():
    args = parse_args()
    directions = [args.direction] if args.direction else ["inbound", "outbound"]
    for d in directions:
        print(f"\n=== Dec 2025 Forecast: {d} ===", flush=True)
        run_direction(d)


if __name__ == "__main__":
    main()
```

### Task 2: Create `run_forecast_2026.py`

**Files:**
- Create: `scripts/run_forecast_2026.py`

**Step 1: Write the script**

This script retrains the best horserace model per market on all data (through Nov 2025) and forecasts Jan–Dec 2026. Structure mirrors `run_horserace.py` but:
- Only processes GPRS + MOC
- Only trains the winning model per series (not all 5)
- Horizon = 12 (full year 2026)
- No actuals column (future data)
- LightGBM needs special handling for future months (iterative 1-step)

```python
#!/usr/bin/env python3
"""2026 Forecast — retrain best horserace model per market, forecast 12 months.

For each market, reads the horserace winner, retrains that single model on
all available data (through Nov 2025), and produces Jan–Dec 2026 forecasts.

LightGBM winners use iterative 1-step-ahead forecasting (feed each prediction
back as a lag for the next month) since future rows don't exist in the data.

Output (to reports/):
  {direction}_forecast_2026.csv

Usage:
  python scripts/run_forecast_2026.py --direction inbound
  python scripts/run_forecast_2026.py --direction outbound
  python scripts/run_forecast_2026.py  # both
"""
from __future__ import annotations

import argparse
import logging
import os
import time
import warnings
from pathlib import Path

logging.disable(logging.CRITICAL)
os.environ["CMDSTAN_VERBOSE"] = "false"

import numpy as np
import pandas as pd

from config import (
    DASHBOARD_CALL_TYPES,
    grain_cols_for, target_for,
)
from models import (
    forecast_seasonal_naive, forecast_ets, forecast_sarima, forecast_theta,
    build_lgbm_features, LGBM_BASE_FEATURES, LGBM_TREND_FEATURES,
)

ROOT = Path(__file__).resolve().parent.parent
HORIZON = 12  # Jan–Dec 2026
FORECAST_MONTHS = list(range(202601, 202613))  # 202601..202612


def parse_args() -> argparse.Namespace:
    p = argparse.ArgumentParser(description="2026 forecast using best horserace model.")
    p.add_argument("--direction", choices=["inbound", "outbound"],
                   default=None, help="Direction (omit for both)")
    return p.parse_args()


PER_SERIES_MODELS = {
    "seasonal_naive": lambda train, h: forecast_seasonal_naive(train, h),
    "ets_damped": lambda train, h: forecast_ets(train, h, damped=True, trend="add"),
    "sarima": lambda train, h: forecast_sarima(train, h),
    "theta": lambda train, h: forecast_theta(train, h),
}


def forecast_lgbm_2026(df: pd.DataFrame, gc: list[str], direction: str,
                        winner_keys: list[tuple]) -> list[dict]:
    """Forecast 2026 for all lgbm-winning series using iterative 1-step prediction.

    Strategy: append 12 placeholder future rows per series, build features
    month-by-month, predict each month, fill the prediction back as the
    actual for lag computation of the next month.
    """
    try:
        import lightgbm as lgb
    except ImportError:
        print("  LightGBM not available", flush=True)
        return []

    rows = []
    feature_cols = list(LGBM_BASE_FEATURES) + LGBM_TREND_FEATURES

    for ct in DASHBOARD_CALL_TYPES:
        target = target_for(ct, direction)
        ct_keys = [k for k in winner_keys
                   if k[gc.index("CALL_TYPE")] == ct]
        if not ct_keys:
            continue

        ct_df = df[df["CALL_TYPE"] == ct].copy()

        # Append 12 future months per series with NaN target
        future_rows = []
        ct_grouped = {k: g for k, g in ct_df.groupby(gc)}
        for key in ct_keys:
            if key not in ct_grouped:
                continue
            base = {c: key[i] for i, c in enumerate(gc)}
            last_row = ct_grouped[key].iloc[-1]
            for fm in FORECAST_MONTHS:
                frow = {**base, "CALL_YEAR_MONTH": fm, target: np.nan,
                        "YEAR": fm // 100, "MONTH": fm % 100}
                # Copy other numeric columns as NaN
                for col in ct_df.columns:
                    if col not in frow:
                        frow[col] = np.nan
                future_rows.append(frow)

        if not future_rows:
            continue

        extended = pd.concat([ct_df, pd.DataFrame(future_rows)], ignore_index=True)
        extended = extended.sort_values(gc + ["CALL_YEAR_MONTH"]).reset_index(drop=True)

        # Iterative 1-step: for each future month, build features, predict, fill back
        for step, fm in enumerate(FORECAST_MONTHS):
            feat_df = build_lgbm_features(extended, target, grain_cols=gc, add_trend=True)

            train_mask = feat_df["CALL_YEAR_MONTH"] < fm
            pred_mask = feat_df["CALL_YEAR_MONTH"] == fm

            train_feat = feat_df[train_mask].dropna(subset=feature_cols)
            pred_feat = feat_df[pred_mask].copy()

            if len(train_feat) < 100 or len(pred_feat) == 0:
                break

            for col in feature_cols:
                pred_feat[col] = pred_feat[col].fillna(train_feat[col].median())

            dtrain = lgb.Dataset(train_feat[feature_cols], train_feat["y"],
                                 free_raw_data=False)
            params = {
                "objective": "regression", "metric": "mae", "num_leaves": 31,
                "learning_rate": 0.05, "feature_fraction": 0.8,
                "bagging_fraction": 0.8, "bagging_freq": 5, "verbose": -1,
                "n_jobs": -1,
            }
            with warnings.catch_warnings():
                warnings.simplefilter("ignore")
                model = lgb.train(params, dtrain, num_boost_round=300)

            preds = np.clip(model.predict(pred_feat[feature_cols]), 0, None)
            pred_feat["_pred"] = preds

            # Fill predictions back into extended df for next iteration's lags
            for _, prow in pred_feat.iterrows():
                key = tuple(prow[c] for c in gc)
                mask = (extended["CALL_YEAR_MONTH"] == fm)
                for c in gc:
                    mask = mask & (extended[c] == prow[c])
                extended.loc[mask, target] = float(prow["_pred"])

                base = {c: prow[c] for c in gc}
                rows.append({
                    **base,
                    "CALL_YEAR_MONTH": fm,
                    "target": target,
                    "model": "lgbm",
                    "predicted": float(prow["_pred"]),
                })

            print(f"    lgbm {ct} step {step+1}/12 ({fm}): {len(pred_feat)} series", flush=True)

        print(f"  lgbm done for {ct}", flush=True)

    return rows


def run_direction(direction: str):
    gc = grain_cols_for(direction)
    reports = ROOT / "reports"

    # Load winners
    winners = pd.read_csv(reports / f"{direction}_horserace_winners.csv")
    winners = winners[winners["CALL_TYPE"].isin(DASHBOARD_CALL_TYPES)]

    # Load data
    df = pd.read_parquet(ROOT / "data" / f"{direction}_set.parquet")
    df = df[df["CALL_TYPE"].isin(DASHBOARD_CALL_TYPES)]
    df = df.sort_values(gc + ["CALL_YEAR_MONTH"]).reset_index(drop=True)
    grouped = {k: g.sort_values("CALL_YEAR_MONTH") for k, g in df.groupby(gc)}

    all_rows: list[dict] = []
    lgbm_keys: list[tuple] = []

    # Per-series models
    per_series_winners = winners[winners["best_model"] != "lgbm"]
    n = len(per_series_winners)
    t0 = time.time()
    fallbacks = 0

    # Also collect fallback models (e.g. sarima_fb → use sarima)
    for i, (_, w) in enumerate(per_series_winners.iterrows()):
        key = tuple(w[c] for c in gc)
        target = w["target"]
        model_name = w["best_model"]

        # Strip _fb suffix to get base model
        base_model = model_name.replace("_fb", "")

        if base_model == "lgbm":
            lgbm_keys.append(key)
            continue

        if key not in grouped:
            continue

        series = grouped[key]
        train_vals = series[target].values.astype(float)

        forecast_fn = PER_SERIES_MODELS.get(base_model)
        if forecast_fn is None:
            # Unknown model, fallback
            fc = forecast_seasonal_naive(train_vals, HORIZON)
            used = f"{model_name}_fb"
            fallbacks += 1
        else:
            fc = forecast_fn(train_vals, HORIZON)
            used = model_name
            if fc is None:
                fc = forecast_seasonal_naive(train_vals, HORIZON)
                used = f"{base_model}_fb"
                fallbacks += 1

        base = {c: w[c] for c in gc}
        for j in range(HORIZON):
            all_rows.append({
                **base,
                "CALL_YEAR_MONTH": FORECAST_MONTHS[j],
                "target": target,
                "model": used,
                "predicted": float(fc[j]),
            })

        if (i + 1) % 200 == 0:
            el = time.time() - t0
            r = (i + 1) / el
            print(f"  [per-series] {i+1}/{n} ({el:.0f}s, {r:.1f}/s)", flush=True)

    el = time.time() - t0
    print(f"  [per-series] {len(per_series_winners)} series in {el:.0f}s ({fallbacks} fallbacks)", flush=True)

    # LightGBM winners
    lgbm_winners = winners[winners["best_model"] == "lgbm"]
    lgbm_key_list = [tuple(row[c] for c in gc) for _, row in lgbm_winners.iterrows()]
    if lgbm_key_list:
        print(f"\n  [lgbm] {len(lgbm_key_list)} series", flush=True)
        lgbm_rows = forecast_lgbm_2026(df, gc, direction, lgbm_key_list)
        all_rows.extend(lgbm_rows)

    # Save
    out_path = reports / f"{direction}_forecast_2026.csv"
    pd.DataFrame(all_rows).to_csv(out_path, index=False)
    print(f"\n  Saved: {out_path} ({len(all_rows)} rows)", flush=True)


def main():
    args = parse_args()
    directions = [args.direction] if args.direction else ["inbound", "outbound"]
    for d in directions:
        print(f"\n=== 2026 Forecast: {d} ===", flush=True)
        run_direction(d)


if __name__ == "__main__":
    main()
```
